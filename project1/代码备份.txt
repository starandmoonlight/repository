        AOS.init({
            duration: 1000,
            once: true,
            offset: 100
        });

        // 添加鼠标跟随效果
        const cursor = document.createElement('div');
        cursor.classList.add('cursor-glow');
        document.body.appendChild(cursor);

        document.addEventListener('mousemove', (e) => {
            cursor.style.transform = `translate(${e.clientX - 10}px, ${e.clientY - 10}px)`;
        });

        // 为可交互元素添加鼠标效果
        const interactiveElements = document.querySelectorAll('a, .feature-card');
        
        interactiveElements.forEach(element => {
            element.addEventListener('mouseenter', () => {
                cursor.style.transform = `translate(${e.clientX - 10}px, ${e.clientY - 10}px) scale(2)`;
            });
            
            element.addEventListener('mouseleave', () => {
                cursor.style.transform = `translate(${e.clientX - 10}px, ${e.clientY - 10}px) scale(1)`;
            });
        });
